(fmtNum et nn inchangés) ... function injecterDonnees(html, rec) { var parser = new DOMParser(); var doc = parser.parseFromString(html, 'text/html'); function set(id, val) { var el = doc.getElementById(id); if (el) el.textContent = val; } var mh = parseFloat(rec.Remun_Init_MontantH) || 0; var hPrev = parseFloat(rec.Remun_Init_Heures) || 0; var a5h = parseFloat(rec.Art5_Heures) || 0; var a5mh = parseFloat(rec.Art5_MontantH) || 0; var today = rec.Date_Signature || new Date().toLocaleDateString('fr-FR'); set('sf-num', nn(rec.Convention_ID)); set('sf-intervenant', nn(rec.Intervenant_Nom)); set('sf-residence', nn(rec.Intervenant_Lieu)); set('sf-intitule', nn(rec.Formation_Intitule)); set('sf-dates', nn(rec.Formation_Dates) + (rec.Formation_Horaires ? ' ' + rec.Formation_Horaires : '')); set('sf-lieu', nn(rec.Formation_Lieu)); set('sf-mh', fmtNum(mh)); set('sf-h-prev', hPrev || '–'); set('sf-h-real', hPrev || '–'); set('sf-st1', fmtNum(rec.Remun_Init_STotal != null ? rec.Remun_Init_STotal : mh * hPrev)); set('sf-art5-h', a5h || '–'); set('sf-art5-mh', fmtNum(a5mh)); set('sf-st2', fmtNum(rec.Art5_STotal != null ? rec.Art5_STotal : a5h * a5mh)); set('sf-creation', rec.Supports_Creation ? fmtNum(rec.Supports_Creation) : '–'); set('sf-maj', rec.Supports_Maj ? fmtNum(rec.Supports_Maj) : '–'); set('sf-supp', nn(rec.Supports_Particuliers_H)); set('sf-st3', fmtNum(rec.Supports_STotal)); set('sf-total', fmtNum(rec.Total_General)); set('sf-date-sig', today); // ── Récupération du signataire via la table Signataires ── var signataire = signatairesParId[rec.ENG_Signataire] || {}; set('sf-identite', nn(signataire.Nom)); set('sf-fonction', nn(signataire.Fonction)); set('sf-service', nn(signataire.Service)); var loading = doc.getElementById('loading'); var contenu = doc.getElementById('contenu'); if (loading) loading.style.display = 'none'; if (contenu) contenu.style.display = ''; var scripts = doc.querySelectorAll('script'); scripts.forEach(function(s) { if (s.textContent.indexOf('grist.onRecord') !== -1) s.remove(); }); return '' + doc.documentElement.outerHTML; } // ... (ouvrirApercu, fermerApercu, imprimerIframe inchangés) ... function verifierPret() { if (htmlModele && currentRecord && signatairesCharges) { document.getElementById('etat').style.display = 'none'; document.getElementById('btn-ouvrir').style.display = 'block'; } else if (htmlModele && !currentRecord) { document.getElementById('etat').textContent = '👆 Sélectionnez une convention dans la table.'; } } if (window.grist) { grist.ready({ requiredAccess: 'read table' }); grist.docApi.fetchTable('Modeles').then(function(table) { if (!table || !table.HTML || table.HTML.length < 2) { document.getElementById('etat').textContent = '⚠️ La table Modeles doit avoir au moins 2 lignes (ligne 2 = Service Fait).'; return; } htmlModele = table.HTML[1] || null; if (!htmlModele) { document.getElementById('etat').textContent = '⚠️ Colonne HTML vide sur la ligne 2 de Modeles.'; return; } verifierPret(); }).catch(function() { document.getElementById('etat').textContent = '⚠️ Table "Modeles" introuvable.'; }); // ── Chargement de la table Signataires ── grist.docApi.fetchTable('Signataires').then(function(table) { var ids = table.id; for (var i = 0; i < ids.length; i++) { signatairesParId[ids[i]] = { Nom: table.Nom[i], Fonction: table.Fonction[i], Service: table.Service[i] }; } signatairesCharges = true; verifierPret(); }).catch(function(e) { document.getElementById('etat').textContent = '⚠️ Table "Signataires" introuvable.'; }); grist.onRecord(function(record) { currentRecord = record; verifierPret(); }); } else { document.getElementById('etat').textContent = '⚠️ Hors Grist — ouvrez ce widget depuis La Suite.'; }
